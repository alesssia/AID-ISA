103  static float vect_variance(floatvect v)
104  {
105  float mean = vect_mean(v);
106  float sum = 0.0;
107  for(unsigned int i=0; i<v.size(); i++)
108  sum += pow((v[i] - mean), 2);
109 
110  return sum/(v.size()-1);
111  }
